Base URL

https://your-domain.com

Authentication

This API does not require authentication. All endpoints are public.

Response Format

All responses are returned in JSON format.

{
  "success": true,
  "data": { ... },
  "checkedAt": "2024-01-01T00:00:00.000Z"
}

Rate Limiting

100 requests per 15 minutes per IP address.

Endpoints

GET /api/check/:username

Check Minecraft account information by username. Returns UUID, username history, and skin data.

Parameters

Name Type Description
username string Minecraft username (1-16 characters)

Example Request

curl -X GET https://your-domain.com/api/check/Obama_Snadwitch

Example Response

{
  "success": true,
  "data": {
    "uuid": "6fa0b728-0989-4b91-9576-845aaca97222",
    "username": "Obama_Snadwitch",
    "username_history": [
      { "username": "Obama_Snadwitch" }
    ],
    "textures": {
      "custom": true,
      "slim": false,
      "skin": {
        "url": "http://textures.minecraft.net/texture/..."
      }
    }
  },
  "checkedAt": "2024-01-01T00:00:00.000Z"
}

  "author": "Rikixz",

GET /api/docs

Get API documentation in JSON format.

GET /api/postman

Get Postman collection for importing into Postman app.

// JavaScript / Node.js
const axios = require('axios');

async checkAccount(username) {
  try {
    const response = await axios.get(
      `https://your-domain.com/api/check/${username}`
    );
    return response.data;
  } catch (error) {
    return { error: error.message };
  }
}

// Usage
checkAccount('Obama_Snadwitch')
  .then(console.log);
# Python
import requests

def check_account(username):
    try:
        response = requests.get(
            f"https://your-domain.com/api/check/{username}"
        )
        return response.json()
    except Exception as e:
        return {"error": str(e)}

# Usage
result = check_account("Obama_Snadwitch")
print(result)
# cURL
curl -X GET https://your-domain.com/api/check/Obama_Snadwitch
// Go
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
)

func checkAccount(username string) map[string]interface{} {
    url := fmt.Sprintf("https://your-domain.com/api/check/%s", username)
    resp, err := http.Get(url)
    if err != nil {
        return map[string]interface{"error": err.Error()}
    }
    defer resp.Body.Close()

    var result map[string]interface{}
    json.NewDecoder(resp.Body).Decode(&result)
    return result
}

func main() {
    fmt.Println(checkAccount("Obama_Snadwitch"))
}
